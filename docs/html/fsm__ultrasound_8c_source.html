210static void do_set_distance(fsm_t *p_this)
211{
212 fsm_ultrasound_t *p_fsm = (fsm_ultrasound_t *)(p_this);
213 uint32_t echo_overflows = port_ultrasound_get_echo_overflows(p_fsm->ultrasound_id);
214 uint32_t echo_init_tick = port_ultrasound_get_echo_init_tick(p_fsm->ultrasound_id);
215 uint32_t echo_end_tick = port_ultrasound_get_echo_end_tick(p_fsm->ultrasound_id);
216 uint32_t time = (echo_end_tick + echo_overflows * 65536 - echo_init_tick);
217 uint32_t distance = time * SPEED_OF_SOUND_MS / 20000;
218 p_fsm->distance_arr[p_fsm->distance_idx] = distance;
219 if (p_fsm->distance_idx >= FSM_ULTRASOUND_NUM_MEASUREMENTS - 1)
220 {
221 qsort(p_fsm->distance_arr, FSM_ULTRASOUND_NUM_MEASUREMENTS, sizeof(uint32_t), _compare);
222 p_fsm->distance_cm = p_fsm->distance_arr[FSM_ULTRASOUND_NUM_MEASUREMENTS / 2]; // Esta es la mediana porque hay un numero IMPAR de elementos
223 p_fsm->new_measurement = true;
224 }
225 p_fsm->distance_idx += 1;
226 if (p_fsm->distance_idx >= FSM_ULTRASOUND_NUM_MEASUREMENTS)
227 p_fsm->distance_idx = 0;
228 port_ultrasound_stop_echo_timer(p_fsm->ultrasound_id);
229 port_ultrasound_reset_echo_ticks(p_fsm->ultrasound_id);
230}
