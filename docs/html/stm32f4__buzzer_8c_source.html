88void _timer_pwm_buzzer_config(uint32_t buzzer_id)
89{

91 RCC->APB1ENR |= RCC_APB1ENR_TIM5EN;
92 /*Segundo, inhabilitamos el contador y habilitamos el autoreload preload.*/
93 TIM5->CR1 &= ~TIM_CR1_CEN;
94 TIM5->CR1 |= TIM_CR1_ARPE;
95 /*Tercero, reseteamos el contador*/
96 TIM5->CNT = 0;
97 /*Cuarto, calculamos ARR y PSC para una frecuencia de 4 kHz, que equivale a un periodo de 0,25 ms.*/
98 double reloj = (double)SystemCoreClock;
99 double periodo = 0.00025;
100 double arr = 65535.0;
101 double psc = round((periodo * reloj / (arr + 1.0)) - 1.0);
102 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
103 if (arr > 65535.0)
104 {
105 psc += 1.0;
106 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
107 }
108 TIM5->ARR = (uint32_t)arr;
109 TIM5->PSC = (uint32_t)psc;
110 /*Quinto, inhabilitamos la comparaci√≥n de salida (output compare).*/
111 TIM5->CCER &= ~TIM_CCER_CC1E;
112 /*Sexto, limpiamos los bits P y NP del Output Compare Register.*/
113 TIM5->CCER &= ~TIM_CCER_CC1P;

137void port_buzzer_set_sound(uint32_t buzzer_id, uint8_t sound)
138{
139 if (buzzer_id == PORT_PARKING_BUZZER_ID)
140 {
141 TIM5->CR1 &= ~TIM_CR1_CEN;
142 if (sound == 0)
143 {
144 /*Inhabilitamos output compare.*/
145 TIM5->CCER &= ~TIM_CCER_CC1E;
146 }
147 else
148 {
149 /*Configuramos el duty cycle.*/
150 uint32_t pwm = (PORT_BUZZER_MAX_VALUE + 1) / 2;
151 TIM5->CCR1 = pwm;
152 /*Cambiamos el periodo para una frecuencia inversamente proporcional a la distancia desde 200 Hz a 2495 Hz*/
153 double reloj = (double)SystemCoreClock;
154 double periodo = 1/(200.0 + 9 * ((double) sound));
155 double arr = 65535.0;
156 double psc = round((periodo * reloj / (arr + 1.0)) - 1.0);
157 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
158 if (arr > 65535.0)
159 {
160 psc += 1.0;
161 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
162 }

164 TIM5->PSC = (uint32_t)psc;
