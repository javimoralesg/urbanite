69void TIM2_IRQHandler(void)
70{
71 /* ISR ultrasound echo timer*/
72 port_system_systick_resume();
73 if (TIM2->SR & TIM_SR_UIF)
74 {
75 uint32_t overflows = port_ultrasound_get_echo_overflows(PORT_REAR_PARKING_SENSOR_ID);
76 overflows++;
77 port_ultrasound_set_echo_overflows(PORT_REAR_PARKING_SENSOR_ID, overflows);
78 overflows = port_ultrasound_get_echo_overflows(PORT_FRONT_PARKING_SENSOR_ID);
79 overflows++;
80 port_ultrasound_set_echo_overflows(PORT_FRONT_PARKING_SENSOR_ID, overflows);
81 TIM2->SR &= ~TIM_SR_UIF;
82 }
83
84 if (TIM2->SR & TIM_SR_CC2IF)
85 {
86 uint32_t currentTicks = TIM2->CCR2;
87 uint32_t init = port_ultrasound_get_echo_init_tick(PORT_REAR_PARKING_SENSOR_ID);
88 uint32_t end = port_ultrasound_get_echo_end_tick(PORT_REAR_PARKING_SENSOR_ID);
89 if (init == 0 && end == 0)
90 {
91 port_ultrasound_set_echo_init_tick(PORT_REAR_PARKING_SENSOR_ID, currentTicks);
92 }
93 else
94 {
95 port_ultrasound_set_echo_end_tick(PORT_REAR_PARKING_SENSOR_ID, currentTicks);
96 port_ultrasound_set_echo_received(PORT_REAR_PARKING_SENSOR_ID, true);
97 }
98 }
99 if (TIM2->SR & TIM_SR_CC1IF)
100 {
101 uint32_t currentTicks = TIM2->CCR1;
102 uint32_t init = port_ultrasound_get_echo_init_tick(PORT_FRONT_PARKING_SENSOR_ID);
103 uint32_t end = port_ultrasound_get_echo_end_tick(PORT_FRONT_PARKING_SENSOR_ID);
104 if (init == 0 && end == 0)
105 {
106 port_ultrasound_set_echo_init_tick(PORT_FRONT_PARKING_SENSOR_ID, currentTicks);
107 }
108 else
109 {
110 port_ultrasound_set_echo_end_tick(PORT_FRONT_PARKING_SENSOR_ID, currentTicks);
111 port_ultrasound_set_echo_received(PORT_FRONT_PARKING_SENSOR_ID, true);
112 }
113 }
114}
