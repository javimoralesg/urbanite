163 /*Quinto, calculamos ARR y PSC para que el periodo del timer sea 10 microsegundos*/
164 double reloj = (double) SystemCoreClock;
165 double periodo = (double) PORT_PARKING_SENSOR_TRIGGER_UP_US;
166 periodo /= 1000000.0;
167 double arr = 65535.0;
168 double psc = round((periodo * reloj / (arr + 1.0)) - 1.0);
169 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
170 if (arr > 65535.0) {
171 psc += 1.0;
172 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
173 }
174 /*Sexto, cargamos ARR y PSC en sus correspondientes registros*/
175 TIM13->ARR = (uint32_t)arr;
176 TIM13->PSC = (uint32_t)psc;
177 /*Septimo, generamos un evento de actualizacion*/
178 TIM13->EGR |= TIM_EGR_UG;
179 /*Octavo, limpiamos las interrupciones*/
180 TIM13->SR &= ~TIM_SR_UIF;
181 /*Noveno, habilitamos las interrupciones del timer*/
182 TIM13->DIER |= TIM_DIER_UIE;

230static void _timer_echo_setup(uint32_t ultrasound_id)
231{
232 /*Primero, habilitamos el timer del echo y deshabilitamos el contador*/
233 RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
234 TIM2->CR1 &= ~TIM_CR1_CEN;
235 /*Segundo, calculamos ARR y PSC para que el periodo del timer sea 1 microsegundo*/
236 double reloj = (double) SystemCoreClock;
237 double periodo = 0.000001;
238 double arr = 65535.0; // Se dice que arr debe configurarse a su m√°ximo
239 double psc = (reloj*periodo) - 1;
240 TIM2->ARR = (uint32_t)arr;
241 TIM2->PSC = (uint32_t)psc;
242 /*Tercero, habilitamos el autoreload preload y generamos un evento de actualizacion*/
243 TIM2->CR1 |= TIM_CR1_ARPE;
244 TIM2->EGR |= TIM_EGR_UG;
245
246 // Configuramos CH2
247 /*Cuarto, marcamos la direccion como input en el registros de Captura/Compare*/
248 TIM2->CCMR1 |= (0x1 << TIM_CCMR1_CC2S_Pos);
249 /*Quinto, deshabilitamos el filtro digital*/
250 TIM2->CCMR1 &= ~TIM_CCMR1_IC2F;
251 /*Sexto, habilitamos la captura de entrada en ambos flancos (subida y bajada), mediante los bits CC2NP y CC2P del registro CCER*/
252 TIM2->CCER |= TIM_CCER_CC2NP;
253 TIM2->CCER |= TIM_CCER_CC2P;
254 /*Septimo, programamos el preescalado de entrada para capturar cada transicion valida, poner a 0*/
255 TIM2->CCMR1 &= ~(TIM_CCMR1_IC2PSC);
256 /*Octavo, habilitamos la captura de entrada*/
257 TIM2->CCER |= TIM_CCER_CC2E;
258 /*Noveno, habilitamos las interrupciones de la captura de entrada*/
259 TIM2->DIER |= TIM_DIER_CC2IE;
260
261 // Configuramos CH1
262 /*Cuarto, marcamos la direccion como input en el registros de Captura/Compare*/
263 TIM2->CCMR1 |= (0x1 << TIM_CCMR1_CC1S_Pos);
264 /*Quinto, deshabilitamos el filtro digital*/
265 TIM2->CCMR1 &= ~TIM_CCMR1_IC1F;
266 /*Sexto, habilitamos la captura de entrada en ambos flancos (subida y bajada), mediante los bits CC2NP y CC2P del registro CCER*/
267 TIM2->CCER |= TIM_CCER_CC1NP;
268 TIM2->CCER |= TIM_CCER_CC1P;
269 /*Septimo, programamos el preescalado de entrada para capturar cada transicion valida, poner a 0*/
270 TIM2->CCMR1 &= ~(TIM_CCMR1_IC1PSC);
271 /*Octavo, habilitamos la captura de entrada*/
272 TIM2->CCER |= TIM_CCER_CC1E;
273 /*Noveno, habilitamos las interrupciones de la captura de entrada*/
274 TIM2->DIER |= TIM_DIER_CC1IE;
275
276 // Configuracion comun
277 /*Decimo, actualizamos las interrupciones del timer*/
278 TIM2->DIER |= TIM_DIER_UIE;
279 /*Undecimo, establecemos las prioridades de las interrupciones*/
280 NVIC_SetPriority(TIM2_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 3, 0));
281}

362void port_ultrasound_init(uint32_t ultrasound_id)
363{
364 /* Get the ultrasound sensor */
365 stm32f4_ultrasound_hw_t *p_ultrasound = _stm32f4_ultrasound_get(ultrasound_id);
366
367 /* Trigger pin configuration */
368 p_ultrasound->trigger_ready = true;
369 p_ultrasound->trigger_end = false;
370 stm32f4_system_gpio_config(p_ultrasound->p_trigger_port, p_ultrasound->trigger_pin, STM32F4_GPIO_MODE_OUT, STM32F4_GPIO_PUPDR_NOPULL);
371
372 /* Echo pin configuration */
373 p_ultrasound->echo_alt_fun = STM32F4_AF1;
374 p_ultrasound->echo_received = false;
375 p_ultrasound->echo_init_tick = 0;
376 p_ultrasound->echo_end_tick = 0;
377 p_ultrasound->echo_overflows = 0;
378 stm32f4_system_gpio_config(p_ultrasound->p_echo_port, p_ultrasound->echo_pin, STM32F4_GPIO_MODE_AF, STM32F4_GPIO_PUPDR_NOPULL);
379 stm32f4_system_gpio_config_alternate(p_ultrasound->p_echo_port, p_ultrasound->echo_pin, p_ultrasound->echo_alt_fun);
380
381 /* Configure timers */
382 // Se configuran una sola vez
383 if (ultrasound_id == 0) {
384 _timer_trigger_setup(ultrasound_id);
385 _timer_echo_setup(ultrasound_id);
386 _timer_new_measurement_setup(ultrasound_id);
387 }
388}

501{
502 stm32f4_ultrasound_hw_t *p_ultrasound = _stm32f4_ultrasound_get(ultrasound_id);
503 p_ultrasound->trigger_ready = false;
504 if (ultrasound_id == PORT_REAR_PARKING_SENSOR_ID)
505 {
506 TIM13->CNT = 0;
507 TIM2->CNT = 0;
508 TIM10->CNT = 0;
509 }
510 if (ultrasound_id == PORT_FRONT_PARKING_SENSOR_ID)
511 {
512 TIM14->CNT = 0;
513 TIM2->CNT = 0;
514 TIM6->CNT = 0;
515 }
516 stm32f4_system_gpio_write(p_ultrasound->p_trigger_port, p_ultrasound->trigger_pin, true);
517 if (ultrasound_id == PORT_REAR_PARKING_SENSOR_ID)
518 {
519 NVIC_EnableIRQ(TIM8_UP_TIM13_IRQn);
520 NVIC_EnableIRQ(TIM2_IRQn);
521 NVIC_EnableIRQ(TIM1_UP_TIM10_IRQn);
522
523 TIM13->CR1 |= TIM_CR1_CEN;
524 TIM2->CR1 |= TIM_CR1_CEN;
525 TIM10->CR1 |= TIM_CR1_CEN;
526 }
527 if (ultrasound_id == PORT_FRONT_PARKING_SENSOR_ID)
