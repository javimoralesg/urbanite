75uint8_t _compute_buzzer_levels(int32_t distance_cm)
76{
77 uint8_t sound = 0;
78 if ((distance_cm >= HIGH_DANGER_MIN_CM) && (distance_cm <= OK_MAX_CM))
79 {
80 sound = PORT_BUZZER_MAX_VALUE * (OK_MAX_CM - distance_cm) / (OK_MAX_CM);
81 }
82 else
83 {
84 sound = PORT_BUZZER_MIN_VALUE;
85 }
86 return sound;
87}
