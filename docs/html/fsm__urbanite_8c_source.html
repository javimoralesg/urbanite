251static void do_distance(fsm_t *p_this)
252{
253 fsm_urbanite_t *p_fsm = (fsm_urbanite_t *)(p_this);
254
255 if (!(p_fsm->is_rear)) {

281 fsm_display_set_status(p_fsm->p_fsm_display_front, true);

287 } else {
288 fsm_display_set_status(p_fsm->p_fsm_display_front, false);

326static void do_pause(fsm_t *p_this)
327{

330
331 p_fsm->is_paused = !p_fsm->is_paused;
332
333 if(p_fsm->is_rear) {
334 fsm_display_set_status(p_fsm->p_fsm_display_rear, p_fsm->is_paused);
335 } else {
336 fsm_display_set_status(p_fsm->p_fsm_display_front, p_fsm->is_paused);
337 }
338 fsm_buzzer_set_status(p_fsm->p_fsm_buzzer, p_fsm->is_paused);
339
340 if(p_fsm->is_paused) printf("[URBANITE][%ld] Urbanite system display PAUSE\n", port_system_get_millis());
341 else printf("[URBANITE][%ld] Urbanite system display RESUME\n", port_system_get_millis());
