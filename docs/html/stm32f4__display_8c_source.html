123void _timer_pwm_config(uint32_t display_id)
124{
125 if (display_id == PORT_REAR_PARKING_DISPLAY_ID)
126 {
127 /*Primero, habilitamos la fuente de reloj del temporizador.*/
128 RCC->APB1ENR |= RCC_APB1ENR_TIM4EN;
129 /*Segundo, inhabilitamos el contador y habilitamos el autoreload preload.*/
130 TIM4->CR1 &= ~TIM_CR1_CEN;
131 TIM4->CR1 |= TIM_CR1_ARPE;
132 /*Tercero, reseteamos el contador*/
133 TIM4->CNT = 0;
134 /*Cuarto, calculamos ARR y PSC para una frecuencia de 50 Hz, que equivale a un periodo de 20 ms.*/
135 double reloj = (double)SystemCoreClock;
136 double periodo = 0.02;
137 double arr = 65535.0;
138 double psc = round((periodo * reloj / (arr + 1.0)) - 1.0);
139 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
140 if (arr > 65535.0)
141 {
142 psc += 1.0;
143 arr = round((periodo * reloj / (psc + 1.0)) - 1.0);
144 }
145 TIM4->ARR = (uint32_t)arr;
146 TIM4->PSC = (uint32_t)psc;
147 /*Quinto, inhabilitamos la comparación de salida (output compare) para los tres canales*/
148 TIM4->CCER &= ~TIM_CCER_CC1E;
149 TIM4->CCER &= ~TIM_CCER_CC3E;

227void port_display_init(uint32_t display_id)
228{
229 stm32f4_display_hw_t *p_display = _stm32f4_display_get(display_id);
230 /*Para el rojo*/
231 stm32f4_system_gpio_config(p_display->p_port_red, p_display->pin_red, STM32F4_GPIO_MODE_AF, STM32F4_GPIO_PUPDR_NOPULL);
232 stm32f4_system_gpio_config_alternate(p_display->p_port_red, p_display->pin_red, STM32F4_AF2);
233 /*Para el verde*/
234 stm32f4_system_gpio_config(p_display->p_port_green, p_display->pin_green, STM32F4_GPIO_MODE_AF, STM32F4_GPIO_PUPDR_NOPULL);
235 stm32f4_system_gpio_config_alternate(p_display->p_port_green, p_display->pin_green, STM32F4_AF2);
236 /*Para el azul*/
237 stm32f4_system_gpio_config(p_display->p_port_blue, p_display->pin_blue, STM32F4_GPIO_MODE_AF, STM32F4_GPIO_PUPDR_NOPULL);
238 stm32f4_system_gpio_config_alternate(p_display->p_port_blue, p_display->pin_blue, STM32F4_AF2);
239 /*Finalmente*/
240 _timer_pwm_config(display_id);
241 port_display_set_rgb(display_id, COLOR_OFF);
